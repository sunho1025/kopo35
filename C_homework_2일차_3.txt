#include <stdio.h>

void main()
{
    float score;
    int score_d;
    char grade;
    printf("Enter Score : ");
    while(1)
    {
        if(scanf("%f",&score) == 1) {
            break;
        } else {
            printf("This is not Digt\n");
            printf("Please Again Enter Score\n");
            while(getchar() != '\n');
        }
    }
    score_d = score/10;
    switch(score_d)
    {
        case 10 :
        case 9 :
            grade = 'A';
            score = score * (1.1);
            break;
        case 8 :

            break;

            break;
