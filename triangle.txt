#include <stdio.h>
//입력값(줄수) :4
//   1
//  2 3
// 4 5 6
int input_num() {
	int n;
	printf("원하는 숫자를 입력하세요 >> ");
	scanf("%d", &n);
	return n;
}

void triangle(int num) {
	int i, j, k;
	k = 1;
	for(i = 0; i < num; i++) {
		for (j = 0; j < num - i; j++) {
			printf(" ");
		}
		for (j = 0; j < i + 1 ; j++) {
			printf("%d ", k);
			k++;
		}
		printf("\n");
	}
}

void main() {
	int num = input_num();
	triangle(num);
}
