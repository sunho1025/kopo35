// 2 day homework
#include <stdio.h>

void main()
{
    int a;
    int i;
    int sum;
    int b[3];
    while(1) {
        printf("Please Enter Number : ");
        if (scanf("%d", &a) == 1) {
            if (a < 1000) {
                break;
            } else {
                printf("Please Enter Number smaller than 1,000\n");
            }
        } else {
            printf("Incorrect Input, Please Again Enter Number!\n");
            while(getchar() != '\n');
        }
    }
        
    for(i = 2 ; i >= 0 ; i--)
    {
        b[i] = a % 10;
        sum += a % 10;
        a /= 10;
    }
    if (a % 2 == 1 && a % 3 == 0)
    {
        printf("This is Special Number");
    } else if (sum % 7 == 0)
    {
        printf("This is Special Number");
    } else if (b[0] > b[2]) {
        printf("This is Special Number");
    } else {
        printf("This is not Special Number");
    } 
    
}
