#include <stdio.h>
#include <math.h>
void main()
{
    int num;
    printf("Enter Number : ");
    scanf("%d", &num);
    
    for (int i = 1; i <= num; i++) {
        for (int k = 1; k < num - i + 1; k++) {
            printf(" ");
        }
        for (int j = 0; j < i; j++) {
            printf("%d ", j + 1);
        }
        printf("\n");
    }
}
