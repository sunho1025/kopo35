#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main()
{
    srand(time(NULL));
    int num[7];
    
    //create lotto number
    int i = 0;
    while(i < 7){
        // duplication number search and change
        num[i] = (rand() % 45) + 1;
        for (int j = 0; j < i ; j++){
            if (num[i] == num[j]){
                i--;
            }
        }
        sleep(1);
        
        i++;
    }
    // Sorting - insert
    int change;
    int length = sizeof(num) / sizeof(int);
    for (i = 0 ; i < length ; i++){
        int j = i - 1;
        while((num[j] > num[j + 1]) && (j >= 0)) {
            change = num[j];
            num[j] = num[j + 1];
            num[j + 1] = change;
            j--;
        }
    }
    
    //print
    printf("Lotto Numbers : ");
    for(int i = 0 ; i < 7 ; i++){
        if (i != 6){
            printf("%d ",num[i]);
        }else{
            printf("  Bonus Number : ");
            printf("%d",num[i]);
        }
        
    }
    return 0;
}
