// 4 day homework_2
#include <stdio.h>
void main()
{
    int num;
    
    while(1) {
        printf("Input Number : ");
        if (scanf("%d", &num) == 1) {
             break;
        } else {
            printf("Please Repeat Input Number!\n");
            while(getchar() != '\n');
        }
    }        
    
    for (int i = 1; i <= num; i++) {
        for (int k = 1; k < num - i + 1; k++) {
            printf(" ");
        }
        for (int j = 0; j < i; j++) {
            printf("%d ", j + 1);
        }
        printf("\n");
    }
    
}
