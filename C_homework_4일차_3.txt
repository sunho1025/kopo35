#include <stdio.h>
void main()
{
    int num;
    
    while(1) {
        printf("Input Number : ");
        if (scanf("%d", &num) == 1) {
             break;
        } else {
            printf("Please Repeat Input Number!\n");
            while(getchar() != '\n');
        }
    }        
    
    for (int i = 0 ; i < num ; i++) {
        for (int j = 0 ; j < num - i ; j++) {
            printf("*");
        }
        printf("\n");
    }
}
