{

   int a = 0, b = 0, c = 0;
   while(1){
       printf("Enter Number 'a, b, c' : ");
       if (scanf("%d", &a) == 1 && scanf("%d", &b) == 1 && scanf("%d", &c) == 1){
           break;
       }else{
           printf("Incorrect Number...\n");
           printf("Please again Enter Number!\n");
           while(getchar() != '\n');
       }
    }
   
   printf("Result : (%d + %d) x %d - (%d - %d) / %d = %d", a, b, c, a, b, c, ((a + b) * c) - ((a - b) / c));
}
