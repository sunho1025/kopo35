#include <stdio.h>
void main()
{
    int num_m;
    int num_n;
    
    while(1) {
        printf("Input Number : ");
        if (scanf("%d", &num_m) == 1 && scanf("%d", &num_n)== 1) {
            if (num_m > num_n) {
                printf("The first input number is greater than the second input number.\n");
                printf("Please Repeat Input Number!\n");
            } else {
                break;
            }
        } else {
            printf("Please Repeat Input Number!\n");
            while(getchar() != '\n');
        }
    }        
    
    for (int i = num_m; i < (num_n + 1); i++) {
        if (i % 7 == 0 || (i + 3) % 10 == 0) {
            continue;
        } else if ( i % 5 == 0) {
            printf("%d ", i);
            break;
        } else {
            printf("%d ", i);
        }
    }
}
